<?php
$conn = mysqli_connect("localhost", "root", "", "company_db");
if (!$conn) {
    die("Connection failed: " . mysqli_connect_error());
}
$id = 1;
$name = "John Doe";
$salary = 50000;
$department = "IT";
$sql = "INSERT INTO emp_master (id, name, salary, department) VALUES ('$id', '$name', '$salary', '$department')";
if (mysqli_query($conn, $sql)) {
    echo "Employee data inserted successfully.";
} else {
    echo "Error: " . mysqli_error($conn);
}
mysqli_close($conn);
?>

Q18

    
<?php
$conn = mysqli_connect("localhost", "root", "", "example_db");
$sql = "SELECT Name, Address, Class, Semester FROM std_info";
$result = mysqli_query($conn, $sql);
echo "<table border='1'><tr><th>Name</th><th>Address</th><th>Class</th><th>Semester</th></tr>";
while ($row = mysqli_fetch_assoc($result)) {
    echo "<tr><td>{$row['Name']}</td><td>{$row['Address']}</td><td>{$row['Class']}</td><td>{$row['Semester']}</td></tr>";
}
echo "</table>";
mysqli_close($conn);
?>
